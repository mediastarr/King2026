function moveFocus(dir){ const isPlayer = q('#player').classList.contains('show'); const isMenu = q('#menuOverlay').classList.contains('show'); const container = keyboardOpen? q('#keyboard') : isPlayer? q('#player') : isMenu? q('#menuOverlay') : screens[currentTab]; const items = getFocusable(container); if(!items.length) return; let idx = items.indexOf(currentFocus); if(idx===-1) idx=0; const cols = container.querySelector('.grid')? 2 : 0; if(dir==='right') idx = (idx+1)%items.length; else if(dir==='left') idx = (idx-1+items.length)%items.length; else if(dir==='down'){ if(cols){ const nextIdx = idx + cols; if(nextIdx < items.length){ idx = nextIdx; } else { // At bottom of grid, move to tabs if(!isPlayer &&!isMenu &&!keyboardOpen){ const tabs = getFocusable(q('#tabs')); if(tabs.length) { focusElement(tabs[0]); return; } } idx = Math.min(idx+cols, items.length-1); } } else { idx = (idx+1)%items.length; } } else if(dir==='up'){ if(cols){ const prevIdx = idx - cols; if(prevIdx >= 0){ idx = prevIdx; } else { idx = 0; } } else { const tabs = getFocusable(q('#tabs')); if(tabs.includes(currentFocus) &&!isPlayer &&!isMenu &&!keyboardOpen){ const contentItems = getFocusable(screens[currentTab]); if(contentItems.length){ const tabIdx = tabs.indexOf(currentFocus); const lastRowStart = Math.floor((contentItems.length-1)/2)*2; idx = Math.min(lastRowStart + tabIdx, contentItems.length-1); focusElement(contentItems[idx]); return; } } idx = (idx-1+items.length)%items.length; } } focusElement(items[idx]); }
